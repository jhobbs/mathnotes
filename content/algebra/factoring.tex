\subsubsection{Sum and Difference of Cubes}

\begin{theorem}[Sum and Difference of Cubes]
Sum of cubes:

\[ a^3 + b^3 = (a+b)(a^2 - ab + b^2) \]

Difference of cubes:

\[ a^3 - b^3 = (a-b)(a^2 + ab + b^2) \]
\end{theorem}

\detach

\begin{note}\label{cubes-factoring-mnemonic}
Note that in both cases

\begin{itemize}
\item the first operation on the right side is the same as the operation on the left side
\item the second operation on the right side is the opposite of the operation on the left side
\item the third operation on the right side is always $+$
\end{itemize}
\end{note}
